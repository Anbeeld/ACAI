		if = {
			limit = { 
				is_ai = yes

				NOT = { has_modifier = ai_fix_subject_1 }
			}
			add_modifier = { 
				modifier = ai_fix_subject_1
				days = -1
			}
		}
		if = {
			limit = { 
				is_ai = yes
				is_subject = no
				is_country_type = default
				has_modifier = ai_fix_subject_1
			}
			remove_modifier = ai_fix_subject_1
		}
	}
