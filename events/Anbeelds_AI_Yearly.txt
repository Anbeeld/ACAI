country_event = {
	id = aai_yearly.20
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_ai = yes
		is_country_type = default
	}
	
	immediate = {
		set_variable = {
			which = acai_value_sectors_core_systems
			value = 3
		}
		
		if = {
			limit = {
				OR = {
					has_valid_civic = civic_efficient_bureaucracy
					has_valid_civic = civic_hive_divided_attention
					has_valid_civic = civic_machine_delegated_functions
				}
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 2
			}
		}
		
		if = {
			limit = {
				has_technology = tech_colonial_bureaucracy
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 1
			}
		}
		
		if = {
			limit = {
				has_technology = tech_galactic_bureaucracy
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 1
			}
		}
		
		if = {
			limit = {
				has_technology = tech_repeatable_improved_core_system_cap
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 1
			}
		}
		
		if = {
			limit = {
				has_tradition = tr_expansion_finish
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 2
			}
		}
		
		if = {
			limit = {
				has_ascension_perk = ap_imperial_prerogative
			}
			change_variable = {
				which = acai_value_sectors_core_systems
				value = 5
			}
		}
		
		set_variable = {
			which = acai_value_sectors_systems_amount
			value = 0
		}
		
		every_system_within_border = {
			limit = {
				root = {
					any_planet_within_border = {
						acai_sectors_colony_in_system = yes
					}
				}
			}
			root = {
				change_variable = {
					which = acai_value_sectors_systems_amount
					value = 1
				}
			}
		}
		
		every_planet_within_border = {
			limit = {
				has_planet_flag = acai_sectors_modified_planet
			}
			remove_planet_flag = acai_sectors_modified_planet
			acai_sectors_remove_modifiers = yes
		}
		
		if = {
			limit = {
				check_variable = {
					which = acai_value_sectors_systems_amount
					value > acai_value_sectors_core_systems
				}
			}
			
			every_system_within_border = {
				limit = {
					NOT = { is_same_value = root.capital_scope.solar_system }
					OR = {
						check_variable = {
							which = acai_value_sectors_system_tiles
							value > 0
						}
						root = {
							any_planet_within_border = {
								acai_sectors_colony_in_system = yes
							}
						}
					}
				}
				if = {
					limit = {
						check_variable = {
							which = acai_value_sectors_system_tiles
							value > 0
						}
					}
					set_variable = {
						which = acai_value_sectors_system_tiles
						value = 0
					}
				}
				root = {
					every_planet_within_border = {
						limit = {
							acai_sectors_colony_in_system = yes
						}
						every_tile = {
							limit = {
								has_grown_pop = yes
							}
							prevprevprev = {
								change_variable = {
									which = acai_value_sectors_system_tiles
									value = 1
								}
							}
						}
					}
				}
				#log = "[root.GetName] system size [this.acai_value_sectors_system_tiles]"
			}
			#every_planet_within_border = {
			#	limit = {
			#		has_modifier = acai_sectors_simulating_losses
			#	}
			#	remove_modifier = acai_sectors_simulating_losses
			#}
			
			subtract_variable = {
				which = acai_value_sectors_systems_amount
				value = acai_value_sectors_core_systems
			}
			
			#log = "[root.GetName] count [this.acai_value_sectors_systems_amount]"
			
			while = {
				count = acai_value_sectors_systems_amount
				random_system_within_border = {
					limit = {
						check_variable = {
							which = acai_value_sectors_system_tiles
							value > 0
						}
						NOT = {
							root = {
								any_planet_within_border = {
									acai_sectors_colony_in_system = yes
									has_planet_flag = acai_sectors_modified_planet
								}
							}
						}
						NOT = {
							root = {
								any_system_within_border = {
									check_variable = {
										which = acai_value_sectors_system_tiles
										value > 0
									}
									NOT = {
										root = {
											any_planet_within_border = {
												acai_sectors_colony_in_system = yes
												has_planet_flag = acai_sectors_modified_planet
											}
										}
									}
									check_variable = {
										which = acai_value_sectors_system_tiles
										value < prevprev
									}
								}
							}
						}
					}
					#log = "[root.GetName] apply [this.acai_value_sectors_system_tiles]"
					root = {
						every_planet_within_border = {
							limit = {
								acai_sectors_colony_in_system = yes
							}
							set_planet_flag = acai_sectors_modified_planet
							acai_sectors_resources_calculating = yes
							acai_sectors_apply_modifiers = yes
						}
					}
				}
			}
		}
	}
}

# Anbeeld's AI: Ships
country_event = {
	id = aai_yearly.10
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_country_type = default
	}
	
	immediate = {
		if = {
			limit = {
				is_ai = yes
				NOR = {
					has_country_flag = acai_corvette_preference
					#has_country_flag = acai_destroyer_preference
					has_country_flag = acai_battleship_preference
				}
			}
			if = {
				limit = {
					has_country_flag = acai_destroyer_preference
				}
				remove_country_flag = acai_destroyer_preference
			}
			random_list = {
				30 = {
					set_country_flag = acai_corvette_preference
					log = "[This.GetName] want corvs"
				}
				#30 = {
				#	set_country_flag = acai_destroyer_preference
				#	log = "[This.GetName] want dstr"
				#}
				30 = {
					set_country_flag = acai_battleship_preference
					log = "[This.GetName] want bs"
				}
			}
		}
		if = {
			limit = {
				is_ai = yes
				NOT = { has_modifier = acai_upgrading_band_aid }
			}
			add_modifier = {
				modifier = acai_upgrading_band_aid
				days = -1
			}
		}
		else_if = {
			limit = {
				is_ai = no
				has_modifier = acai_upgrading_band_aid
			}
			remove_modifier = acai_upgrading_band_aid
		}
	}
}

country_event = {
	id = aai_yearly.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_ai = yes
		is_country_type = default
		minerals > 100
		OR = {
			AND = {
				years_passed < 5
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_constructor
					}
					count < 1
				}
			}
			AND = {
				years_passed >= 5
				years_passed < 10
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_constructor
					}
					count < 2
				}
			}
			AND = {
				years_passed >= 10
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_constructor
					}
					count < 3
				}
			}
			AND = {
				years_passed < 5
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_science_ship
					}
					count < 1
				}
			}
			AND = {
				years_passed >= 5
				years_passed < 10
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_science_ship
					}
					count < 2
				}
			}
			AND = {
				years_passed >= 10
				count_owned_ships = {
					limit = {
						is_ship_class = shipclass_science_ship
					}
					count < 3
				}
			}
		}
	}
	
	immediate = {
		country_event = { id = aai_civilian.1 }
	}
}
