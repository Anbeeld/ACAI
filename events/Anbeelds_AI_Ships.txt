country_event = {
	id = aai_ships.1
	hide_window = yes
	is_triggered_only = yes
	
	# Choosing where and what to build
	immediate = {
	
	##log = "[this.GetName] ships processing"
		if = {
			limit = {
				any_country = {
					is_at_war_with = root
					relative_power = {
						who = root
						category = fleet
						value >= superior
					}
				}
			}
			if = {
				limit = {
					aai_energy_income_low = no
					acai_minerals_income_extreme_margin = no
				}
				acai_ships_build = yes
			}
			else_if = {
				limit = {
					acai_minerals_income_extreme = yes
				}
				acai_ships_destroy = yes
			}
		}
		else_if = {
			limit = {
				any_country = {
					acai_is_rival = yes
					relative_power = {
						who = root
						category = fleet
						value >= overwhelming
					}
				}
			}
			if = {
				limit = {
					aai_energy_income_low = no
					acai_minerals_income_low_margin = no
				}
				acai_ships_build = yes
			}
			else_if = {
				limit = {
					acai_minerals_income_low = yes
				}
				acai_ships_destroy = yes
			}
		}
		else_if = {
			limit = {
				any_country = {
					OR = {
						acai_is_rival = yes
						is_at_war_with = root
					}
					relative_power = {
						who = prev
						category = fleet
						value >= equivalent
					}
				}
			}
			if = {
				limit = {
					acai_naval_capacity_free_high = yes
					aai_energy_income_low = no
					acai_minerals_income_low_margin = no
				}
				acai_ships_build = yes
			}
			else_if = {
				limit = {
					is_at_war = yes
					OR = {
						acai_naval_capacity_delete_high = yes
						acai_minerals_income_extreme = yes
					}
				}
				acai_ships_destroy = yes
			}
			else_if = {
				limit = {
					is_at_war = no
					OR = {
						acai_naval_capacity_delete_high = yes
						acai_minerals_income_low = yes
					}
				}
				acai_ships_destroy = yes
			}
		}
		else_if = {
			limit = {
				OR = {
					any_country = {
						has_defensive_pact = root
					}
					has_federation = yes
					AND = {
						any_neighbor_country = {
							acai_is_rival = yes
						}
						NOT = {
							any_neighbor_country = {
								acai_is_rival = yes
								relative_power = {
									who = prev
									category = fleet
									value >= equivalent
								}
							}
						}
					}
					is_at_war = yes
				}
			}
			if = {
				limit = {
					acai_naval_capacity_free_medium = yes
					aai_energy_income_low = no
					acai_minerals_income_low_margin = no
				}
				acai_ships_build = yes
			}
			else_if = {
				limit = {
					is_at_war = yes
					OR = {
						acai_naval_capacity_delete_medium = yes
						acai_minerals_income_low = yes
					}
				}
				acai_ships_destroy = yes
			}
			else_if = {
				limit = {
					is_at_war = no
					OR = {
						acai_naval_capacity_delete_medium = yes
						acai_minerals_income_low = yes
					}
				}
				acai_ships_destroy = yes
			}
		}
		else = {
			if = {
				limit = {
					acai_naval_capacity_free_low = yes
					aai_energy_income_low = no
					acai_minerals_income_low_margin = no
				}
				acai_ships_build = yes
			}
			else_if = {
				limit = {
					OR = {
						acai_naval_capacity_delete_low = yes
						acai_minerals_income_low = yes
					}
				}
				acai_ships_destroy = yes
			}
		}
	}
}
