acai_rogue_servitors_not_organic = {
	OR = {
		owner = {
			NOT = { has_civic = civic_machine_servitor }
		}
		AND = {
			owner = {
				has_civic = civic_machine_servitor
			}
			pop = {
				is_robot_pop = yes
			}
		}
	}
}

acai_buildings_empty_tile_conditions = {
	aai_suitable_planet_for_construction = yes
	any_tile = {
		has_building = no
		has_blocker = no
		acai_rogue_servitors_not_organic = yes
		OR = {
			has_grown_pop = yes
			has_growing_pop = yes
			owner = {
				minerals > 2000
				NOR = { 
					has_ascension_perk = ap_voidborn
					has_ascension_perk = ap_galactic_wonders
				}
			}
		}
	}
}
