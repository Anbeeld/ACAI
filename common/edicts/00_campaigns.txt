# Sets the edict-cost for all edicts with this variable
@campaignCost = 1000
@campaignDuration = 3600

#################
# Campaigns
# shorter duration, cost energy
#################

country_edict = {
	name = "fear_campaign"
	length = @campaignDuration
	cost = { energy = @campaignCost }
	
	modifier = {
		country_resource_unity_mult = 0.10
		pop_ethic_xenophobe_attraction_mult = 0.25
	}
	
	potential = {
		is_xenophobe = yes
		has_comms_with_alien_civilization = yes
		has_technology = tech_planetary_unification
	}
	
	ai_weight = {
		weight = 20
		modifier = {
			factor = 0
			OR = {
				has_country_resource = {
					type = energy
					amount < 2000
				}
				has_monthly_income = {
					resource = energy
					value < -10
				}
				AND = {
					has_country_resource = {
						type = energy
						amount < 6000
					}
					has_monthly_income = {
						resource = energy
						value < 0
					}
				}
			}
		}
		modifier = {
			factor = 10
			has_country_resource = {
				type = energy
				amount >= 6000
			}
		}
		modifier = {
			factor = 10
			has_country_resource = {
				type = energy
				amount >= 8000
			}
		}
	}	
}
