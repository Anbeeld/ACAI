
acai_starbases_count_by_type = {

	every_system_within_border = {
		limit = {
			starbase = {
				has_starbase_size = starbase_outpost
				acai_starbases_has_no_use = no
			}
		}
		starbase = {
			acai_starbases_clear_use = yes
		}
	}
	
	set_variable = { which = acai_value_starbases_upgraded_amount value = 0 }
	every_owned_starbase = {
		limit = {
			has_starbase_size >= starbase_starport
		}
		owner = {
			change_variable = {
				which = acai_value_starbases_upgraded_amount
				value = 1
			}
		}
	}
	
	set_variable = { which = acai_value_starbases_shipyards_amount value = 0 }
	every_owned_starbase = {
		limit = {
			acai_starbases_is_shipyard = yes
		}
		owner = {
			change_variable = {
				which = acai_value_starbases_shipyards_amount
				value = 1
			}
		}
	}
	
	set_variable = { which = acai_value_starbases_trading_hubs_amount value = 0 }
	every_owned_starbase = {
		limit = {
			acai_starbases_is_trading_hub = yes
		}
		owner = {
			change_variable = {
				which = acai_value_starbases_trading_hubs_amount
				value = 1
			}
		}
	}
	
}
