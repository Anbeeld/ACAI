
acai_healing_remove_going_to_system = {
	if = {
		limit = {
			check_variable = {
				which = acai_boolean_healing_going_to_system
				value = 2
			}
		}
		clear_fleet_actions = this
		clear_orders = yes
		set_variable = {
			which = acai_boolean_healing_going_to_system
			value = 1
		}
	}
}

acai_healing_remove_fleet_locking = {
	if = {
		limit = {
			check_variable = {
				which = acai_boolean_healing_fleet_is_locked
				value = 2
			}
		}
		remove_modifier = acai_healing_locking_by_speed
		remove_modifier = acai_healing_emulate_repairing
		set_variable = {
			which = acai_boolean_healing_fleet_is_locked
			value = 1
		}
	}
}

acai_healing_remove_everything = {
	set_variable = {
		which = acai_boolean_healing_triggered
		value = 1
	}
	acai_healing_remove_going_to_system = yes
	acai_healing_remove_fleet_locking = yes
}
