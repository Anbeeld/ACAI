		which = acai_value_naval_capacity_free_low

			value = acai_value_naval_capacity_ratio_medium_delete_war

		}
		set_variable = {
			which = acai_value_naval_capacity_ratio_medium_delete
			value = acai_value_naval_capacity_ratio_medium_delete_peace
		}

	}
